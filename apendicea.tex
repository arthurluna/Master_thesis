\begin{chapter}{Fator de eficiência e coeficientes de multipolo}
\label{apendicea}

\hspace{5 mm} Nesse apêndice são mostradas as expressões para as componentes cilindricas azimutal e radial do fator de eficiência. Dividimos a força em componentes de extinção e expalhamento, como na seção \ref{forcaotica}. Também são mostrados os coeficientes de multipolo $\G^{(\sigma)}_{j,m}$, $\G^{C,(\sigma)}_{j,m}$ e $\G^{\pm,(\sigma)}_{j,m}$ e as funções $f^{(\sigma)}(\rho,\phi,z)$.

As componentes de extinção da força nas direções $\rho$ e $\phi$ são:

\begin{equation}
\begin{split}
Q_{e\rho}(\rho,\phi,z) = \frac{\gamma^2}{AN}\operatorname{Im}\sum\limits_{jm\sigma} (2j+1)\G^{(\sigma)}_{j,m} \Big{[} (a_j+b_j) \Big{(}\G^{-,(\sigma)*}_{j,m+1}-\G^{+,(\sigma)*}_{j,m-1}\Big{)}(1-\sigma\sin 2\psi) + \\ +(a_j-b_j) \Big{(}\G^{-,(-\sigma)*}_{j,m+1}-\G^{+,(-\sigma)*}_{j,m-1}\Big{)}\cos2\psi \> e^{i2\sigma(\phi-\psi)} \big{]},\qquad \qquad \qquad 
\end{split}
\end{equation}
%
\begin{equation}
\begin{split}
Q_{e\phi}(\rho,\phi,z) = -\frac{\gamma^2}{AN}\operatorname{Re}\sum\limits_{jm\sigma} (2j+1)\G^{(\sigma)}_{j,m} \Big{[} (a_j+b_j) \Big{(}\G^{-,(\sigma)*}_{j,m+1}+\G^{+,(\sigma)*}_{j,m-1}\Big{)}(1-\sigma\sin 2\psi) + \\ +(a_j-b_j) \Big{(}\G^{-,(-\sigma)*}_{j,m+1}+\G^{+,(-\sigma)*}_{j,m-1}\Big{)}\cos2\psi \> e^{i2\sigma(\phi-\psi)} \big{]}. \qquad \qquad \qquad \quad
\end{split}
\end{equation}
%

As componentes de espalhamento da força nas direções $\rho$ e $\phi$ são:

\begin{equation}
\begin{split}
Q_{s\rho}(\rho,\phi,z) = \frac{2\gamma^2}{AN}\operatorname{Im}\sum\limits_{jm\sigma} \Big{[} \frac{\sqrt{j(j+2)(j+m+1)(j+m+2)}}{j+1} \Big{(} (a_ja^*_{j+1}+b_jb^*_{j+1})\times \\ \big{(}\G^{(\sigma)}_{j,m}\G^{(\sigma)*}_{j+1,m+1} + \G^{(\sigma)}_{j,-m}\G^{(\sigma)*}_{j+1,-(m+1)}\big{)}(1-\sigma\sin2\psi) + (a_ja^*_{j+1}-b_jb^*_{j+1})\times \qquad \quad \\ \big{(}\big{(}\G^{(\sigma)}_{j,m}\G^{(-\sigma)*}_{j+1,m+1} + \G^{(\sigma)}_{j,-m}\G^{(-\sigma)*}_{j+1,-(m+1)}\big{)} \cos2\psi e^{i2\sigma(\phi-\psi)} \Big{)} \Big{]}- \qquad \qquad \qquad\\-\frac{4\gamma^2}{AN}\sum\limits_{jm\sigma} \frac{2j+1}{j(j+1)} \sigma \sqrt{(j-m)(j+m+1)}\Big{(}\operatorname{Re}(a_jb^*_j) \operatorname{Im}\big{(}\G^{(\sigma)}_{j,m}\G^{(\sigma)*}_{j,m+1}(1-\sigma\sin2\psi)\big{)} + \\ +\operatorname{Im}(a_jb^*_j)\operatorname{Re}\big{(}\G^{(\sigma)}_{j,m+1}\G^{(-\sigma)*}_{j,m}\cos2\psi e^{i2\sigma(\phi-\psi)}\big{)} \Big{)}, \qquad \qquad \qquad \qquad
\end{split}
\end{equation}
%
\begin{equation}
\begin{split}
Q_{s\phi}(\rho,\phi,z) = -\frac{2\gamma^2}{AN}\operatorname{Re}\sum\limits_{jm\sigma} \Big{[} \frac{\sqrt{j(j+2)(j+m+1)(j-m+1)}}{j+1} \Big{(} (a_ja^*_{j+1}+b_jb^*_{j+1})\times \\ \big{(}\G^{(\sigma)}_{j,m}\G^{(\sigma)*}_{j+1,m+1} - \G^{(\sigma)}_{j,-m}\G^{(\sigma)*}_{j+1,-(m+1)}\big{)}(1-\sigma\sin2\psi) + (a_ja^*_{j+1}-b_jb^*_{j+1})\times \qquad \quad \\ \big{(}\big{(}\G^{(\sigma)}_{j,m}\G^{(-\sigma)*}_{j+1,m+1} - \G^{(\sigma)}_{j,-m}\G^{(-\sigma)*}_{j+1,-(m+1)}\big{)} \cos2\psi \> e^{i2\sigma(\phi-\psi)} \Big{)} \Big{]}+ \qquad \qquad \qquad\\+\frac{4\gamma^2}{AN}\sum\limits_{jm\sigma} \frac{2j+1}{j(j+1)} \sigma \sqrt{(j-m)(j+m+1)}\Big{(}\operatorname{Re}(a_jb^*_j) \operatorname{Re}\big{(}\G^{(\sigma)}_{j,m}\G^{(\sigma)*}_{j,m+1}(1-\sigma\sin2\psi)\big{)} + \\ +\operatorname{Im}(a_jb^*_j)\operatorname{Im}\big{(}\G^{(\sigma)}_{j,m+1}\G^{(-\sigma)*}_{j,m}\cos2\psi \> e^{i2\sigma(\phi-\psi)}\big{)} \Big{)}. \qquad \qquad \qquad \qquad
\end{split}
\end{equation}
%

Os coeficientes de multipolo são:

\begin{equation}
\G^{(\sigma)}_{j,m} (\rho,\phi,z) = \int_0^{\theta_0} d\theta_k \sin\theta_k \sqrt{\cos\theta_k}\>e^{\gamma^2\sin^2\theta_k} T(\theta_k) d^j_{m,\sigma}(\theta_w)f^{(\sigma)}_m(\rho,\phi,z)\>e^{\Phi_{g-w}(\theta_k)+k_wz\cos\theta_w},
\end{equation}
%

\begin{equation}
\G^{C,(\sigma)}_{j,m} (\rho,\phi,z) = \int_0^{\theta_0} d\theta_k \sin\theta_k \cos\theta_w \sqrt{\cos\theta_k}\>e^{\gamma^2\sin^2\theta_k} T(\theta_k) d^j_{m,\sigma}(\theta_w)f^{(\sigma)}_m(\rho,\phi,z)\>e^{\Phi_{g-w}(\theta_k)+k_wz\cos\theta_w},
\end{equation}
%

\begin{equation}
\G^{\pm,(\sigma)}_{j,m} (\rho,\phi,z) = \int_0^{\theta_0} d\theta_k \sin\theta_k \sin\theta_w \sqrt{\cos\theta_k}\>e^{\gamma^2\sin^2\theta_k} T(\theta_k) d^j_{m,\sigma}(\theta_w)f^{(\sigma)}_m(\rho,\phi,z)\>e^{\Phi_{g-w}(\theta_k)+k_wz\cos\theta_w}.
\end{equation}
%

{\bf Falta double check e procurar as definições de cada termo ao longo da dissertação. Falta definir a função $f$}

A função $f^{(\sigma)}(\rho,\phi,z)$ usada nos coeficientes acima são:

\begin{equation}
f^{(\sigma)}(\rho,\phi,z)=\sum\limits_{s=-\infty}^{\infty}(-i)^sJ_s\Big{(}2\pi A_{ast} \frac{\sin^2\theta_k}{\sin^2\theta_0} \Big{)} J_{2s+m-\sigma}(k\rho\sin\theta_k) \>e^{i2s(\varphi_a-\phi)}.
\label{ap_f}
\end{equation}

Nesta função, o parâmetro $A_{ast}$ é o parâmetro de astigmatismo no formalismo de Zernike, que é brevemente discutido no apêndice \ref{apendiceb}. Temos também as funções $J_m$, que são funções de Bessel cilindricas do primeiro tipo. 

Como podemos ver, a equação \ref{ap_f} apresenta um somatório que pode ser desafiador de ser resolver numericamente. Evitamos esse problema restringindo nossos cálculos ao caso em que $\rho=0$, o que faz com que somente um termo da soma não seja igual a 0. Como a única função de Bessel que é diferente de 0 na orígem é $J_0(x)$, o único índice $s$ que corresponde a um termo não nulo é $s=\sigma-m$.

De fato isso não implica em problema algum para as simulações, pois para calcular as posições de equilíbrio, calculamos $Q_z$ em cima do eixo, ou seja, com $\rho=0$. O cálculo das constantes elásticas da pinça também não sofrem com essa condição, pois a derivada de $f^{(\sigma)}(\rho,\phi,z)$ em relação a $\rho$ é calculada na posição de equilíbrio, também com $\rho=0$.

\end{chapter}
